function prep(node, depth = 0, parent = null) {
  if (!node.id) node.id = (parent.id + '-' + node.label).replace(/[^a-z0-9-]+/gi, '-').toLowerCase();
  if (!node.type && parent) node.type = parent.type;
  if (!node.kind) node.kind = depth === 0 ? 'root' : depth === 1 ? 'trunk' : depth === 2 ? 'branch' : 'leaf';
  node._parent = parent;
  node._depth = depth;
  node.collapsed = (depth >= 2);
  if (node.children) node.children.forEach(c => prep(c, depth + 1, node));
}

/* ============================================================
   LAYOUT · radial, leaf-count weighted
   ============================================================ */
const RADII = [0, 320, 580, 850];
const TWO_PI = Math.PI * 2;

function visibleLeafCount(node) {
  if (node.collapsed || !node.children || !node.children.length) return 1;
  return node.children.reduce((s, c) => s + visibleLeafCount(c), 0);
}

function layout(node, a0, a1, depth) {
  const ang = (a0 + a1) / 2;
  const r = RADII[Math.min(depth, RADII.length - 1)];
  node.x = depth === 0 ? 0 : Math.cos(ang - Math.PI / 2) * r;
  node.y = depth === 0 ? 0 : Math.sin(ang - Math.PI / 2) * r;
  node.angle = ang;
  if (!node.collapsed && node.children && node.children.length) {
    const total = visibleLeafCount(node);
    const span = a1 - a0;
    let cur = a0;
    for (const c of node.children) {
      const leaves = visibleLeafCount(c);
      const cs = span * (leaves / total);
      layout(c, cur, cur + cs, depth + 1);
      cur += cs;
    }
  }
}

function createNodeEl(node) {

function radialPath(p, c) {
  const pr = Math.hypot(p.x, p.y) || 1;
  const cr = Math.hypot(c.x, c.y);
  if (pr < 2) {
    return `M ${p.x} ${p.y} L ${c.x} ${c.y}`;
  }
  const mid = (pr + cr) / 2;
  const pa = Math.atan2(p.y, p.x);
  const ca = Math.atan2(c.y, c.x);
  const c1x = Math.cos(pa) * mid, c1y = Math.sin(pa) * mid;
  const c2x = Math.cos(ca) * mid, c2y = Math.sin(ca) * mid;
  return `M ${p.x} ${p.y} C ${c1x} ${c1y}, ${c2x} ${c2y}, ${c.x} ${c.y}`;
}

/* ============================================================
   HOVER · highlight ancestors / descendants / cross-links
   ============================================================ */
function onHover(node, isOn, evt) {
  if (isOn && node.tip) {
    tooltip.textContent = node.tip;
    tooltip.classList.add('show');
    if (evt) {
      tooltip.style.left = (evt.clientX + 14) + 'px';
      tooltip.style.top  = (evt.clientY + 14) + 'px';
    }
  } else {
    tooltip.classList.remove('show');
  }
  if (!isOn) {
    nodeEls.forEach(el => { el.classList.remove('dimmed'); el.classList.remove('highlighted'); });
    svg.querySelectorAll('path').forEach(p => { p.classList.remove('highlighted'); p.classList.remove('dimmed'); });
    return;
  }
  const keep = new Set();
  let n = node; while (n) { keep.add(n.id); n = n._parent; }
  (function down(x) {
    keep.add(x.id);
    if (!x.collapsed && x.children) x.children.forEach(down);
  })(node);
  CROSS_LINKS.forEach(l => {
    if (l.from === node.id) keep.add(l.to);
    if (l.to === node.id) keep.add(l.from);
  });
  nodeEls.forEach((el, id) => {
    if (keep.has(id)) el.classList.add('highlighted');
    else el.classList.add('dimmed');
  });
  svg.querySelectorAll('path').forEach(p => {
    if (keep.has(p.dataset.from) && keep.has(p.dataset.to)) p.classList.add('highlighted');
    else p.classList.add('dimmed');
  });
}
